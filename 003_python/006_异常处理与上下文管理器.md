## 一、Python 的异常处理机制

### 面试问题

Python 的异常处理是如何工作的？`try...except...else...finally` 的执行顺序是什么？

### 标准答案

**基本语法：**

```python
try:
    # 可能抛出异常的代码
    result = 10 / 0
except ZeroDivisionError as e:
    # 捕获并处理特定异常
    print(f"除零错误: {e}")
except (ValueError, TypeError) as e:
    # 捕获多种异常
    print(f"值或类型错误: {e}")
except Exception as e:
    # 捕获所有常规异常（不包括 SystemExit、KeyboardInterrupt）
    print(f"其他错误: {e}")
else:
    # 没有异常时执行
    print("执行成功")
finally:
    # 无论是否异常都执行（资源清理）
    print("总是执行")
```

**执行顺序：**

```python
def demo():
    try:
        print("1. try 块")
        result = 10 / 0
    except ZeroDivisionError:
        print("2. except 块")
    else:
        print("3. else 块（不会执行，因为有异常）")
    finally:
        print("4. finally 块（总是执行）")

demo()
# 1. try 块
# 2. except 块
# 4. finally 块
```

**自定义异常：**

```python
class AppError(Exception):
    """应用基础异常"""
    pass

class ValidationError(AppError):
    """数据验证异常"""
    def __init__(self, field, message):
        self.field = field
        self.message = message
        super().__init__(f"{field}: {message}")

class NotFoundError(AppError):
    """资源未找到异常"""
    def __init__(self, resource, id):
        self.resource = resource
        self.id = id
        super().__init__(f"{resource} with id {id} not found")

# 使用
def create_user(data):
    if not data.get("name"):
        raise ValidationError("name", "名称不能为空")
    if len(data["name"]) > 100:
        raise ValidationError("name", "名称长度不能超过100")

try:
    create_user({"name": ""})
except ValidationError as e:
    print(f"验证失败: {e.field} - {e.message}")
```

**异常链（Exception Chaining）：**

```python
try:
    try:
        1 / 0
    except ZeroDivisionError as e:
        raise ValueError("计算错误") from e  # 保留原始异常
except ValueError as e:
    print(f"异常: {e}")
    print(f"原始异常: {e.__cause__}")
```

**面试加分点：** `raise ... from ...` 可以保留异常链，方便调试。使用 `__cause__` 查看原始异常，`__traceback__` 查看调用栈。

## 二、Python 的上下文管理器

### 面试问题

什么是上下文管理器？如何用 `yield` 简化上下文管理器的实现？

### 标准答案

**基本实现：**

```python
class FileManager:
    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode
        self.file = None

    def __enter__(self):
        """进入 with 语句时调用"""
        self.file = open(self.filename, self.mode)
        return self.file

    def __exit__(self, exc_type, exc_val, exc_tb):
        """离开 with 语句时调用"""
        if self.file:
            self.file.close()
        # 返回 True 会抑制异常，返回 False 会让异常继续传播
        return False

with FileManager("test.txt", "w") as f:
    f.write("hello")
# 文件会自动关闭，即使发生异常
```

**使用 `contextlib.contextmanager`：**

```python
from contextlib import contextmanager

@contextmanager
def file_manager(filename, mode):
    """使用 yield 简化上下文管理器"""
    try:
        f = open(filename, mode)
        yield f  # yield 之前的代码在 __enter__ 中执行
    finally:
        f.close()  # yield 之后的代码在 __exit__ 中执行

with file_manager("test.txt", "w") as f:
    f.write("hello")
```

**常见应用场景：**

```python
from contextlib import contextmanager
import time
import sys
import os

# 1. 计时器
@contextmanager
def timer(label=""):
    start = time.time()
    try:
        yield
    finally:
        elapsed = time.time() - start
        print(f"{label} 耗时: {elapsed:.4f}s")

with timer("数据处理"):
    time.sleep(1)

# 2. 临时修改环境变量
@contextmanager
def env_var(key, value):
    old_value = os.environ.get(key)
    os.environ[key] = value
    try:
        yield
    finally:
        if old_value is None:
            del os.environ[key]
        else:
            os.environ[key] = old_value

with env_var("DEBUG", "true"):
    print(os.environ["DEBUG"])  # true

# 3. 重定向输出
@contextmanager
def redirect_stdout(file):
    old_stdout = sys.stdout
    sys.stdout = file
    try:
        yield
    finally:
        sys.stdout = old_stdout

with open("output.txt", "w") as f:
    with redirect_stdout(f):
        print("这会写入文件")

# 4. 数据库事务
@contextmanager
def transaction(connection):
    try:
        yield connection
        connection.commit()
    except Exception:
        connection.rollback()
        raise

# 5. 临时切换目录
@contextmanager
def cd(path):
    old_dir = os.getcwd()
    os.chdir(path)
    try:
        yield
    finally:
        os.chdir(old_dir)

with cd("/tmp"):
    print(os.getcwd())  # /tmp
print(os.getcwd())  # 原来的目录
```

**`suppress` — 抑制特定异常：**

```python
from contextlib import suppress

# 如果文件不存在，忽略 FileNotFoundError
with suppress(FileNotFoundError):
    os.remove("nonexistent.txt")

# 等价于
try:
    os.remove("nonexistent.txt")
except FileNotFoundError:
    pass
```

**`closing` — 确保对象被关闭：**

```python
from contextlib import closing
import urllib.request

with closing(urllib.request.urlopen("https://example.com")) as page:
    for line in page:
        print(line)
# 即使发生异常，也会调用 page.close()
```

**异步上下文管理器：**

```python
import asyncio

class AsyncTimer:
    async def __aenter__(self):
        self.start = time.time()
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        self.elapsed = time.time() - self.start
        print(f"耗时: {self.elapsed:.4f}s")

async def main():
    async with AsyncTimer():
        await asyncio.sleep(1)

asyncio.run(main())
```

**面试加分点：** 上下文管理器的核心价值是**确保资源清理**（RAII 模式）。即使发生异常，`__exit__` 或 `finally` 块也会执行，避免资源泄漏。
