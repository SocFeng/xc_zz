## 一、Python 的多线程（threading）

### 面试问题

Python 的多线程是如何工作的？有哪些同步原语？什么场景下适合使用多线程？

### 标准答案

**基本用法：**

```python
import threading
import time

def worker(name, delay):
    print(f"线程 {name} 开始")
    time.sleep(delay)
    print(f"线程 {name} 结束")

# 创建线程
t1 = threading.Thread(target=worker, args=("A", 2))
t2 = threading.Thread(target=worker, args=("B", 1))

# 启动线程
t1.start()
t2.start()

# 等待线程结束
t1.join()
t2.join()
print("所有线程完成")
```

**线程同步原语：**

```python
import threading

# 1. Lock（互斥锁）：最基本的锁
lock = threading.Lock()
shared_data = 0

def increment():
    global shared_data
    for _ in range(100000):
        with lock:  # 自动获取和释放锁
            shared_data += 1

threads = [threading.Thread(target=increment) for _ in range(5)]
for t in threads:
    t.start()
for t in threads:
    t.join()
print(shared_data)  # 500000

# 2. RLock（可重入锁）：同一线程可以多次获取
rlock = threading.RLock()

def recursive(n):
    with rlock:
        if n > 0:
            recursive(n - 1)

# 3. Semaphore（信号量）：控制并发数量
semaphore = threading.Semaphore(3)  # 最多 3 个线程同时访问

def limited_access(name):
    with semaphore:
        print(f"{name} 正在访问")
        time.sleep(1)
        print(f"{name} 完成访问")

# 4. Condition（条件变量）：线程间通信
condition = threading.Condition()
items = []

def producer():
    for i in range(5):
        with condition:
            items.append(i)
            condition.notify()  # 通知消费者
        time.sleep(0.5)

def consumer():
    for _ in range(5):
        with condition:
            while not items:
                condition.wait()  # 等待生产者
            item = items.pop(0)
            print(f"消费: {item}")

# 5. Event（事件）：线程间信号
event = threading.Event()

def waiter():
    print("等待事件...")
    event.wait()  # 阻塞直到事件被设置
    print("事件已设置！")

def setter():
    time.sleep(2)
    event.set()  # 设置事件

# 6. Barrier（屏障）：等待所有线程到达
barrier = threading.Barrier(3)

def sync_point(name):
    print(f"{name} 到达屏障")
    barrier.wait()  # 阻塞直到 3 个线程都到达
    print(f"{name} 通过屏障")
```

**线程安全的数据结构：**

```python
import queue

# Queue：线程安全的队列
q = queue.Queue(maxsize=10)
q.put(item)
item = q.get()
q.task_done()

# LifoQueue：后进先出
# PriorityQueue：优先级队列
```

**适用场景：** IO 密集型任务（网络请求、文件读写、数据库查询）。受 GIL 限制，不适合 CPU 密集型任务。

## 二、Python 的多进程（multiprocessing）

### 面试问题

多进程和多线程有什么区别？如何使用多进程绕过 GIL？

### 标准答案

**基本用法：**

```python
from multiprocessing import Process, Pool
import os

def worker(name):
    print(f"进程 {name} (PID: {os.getpid()})")
    return f"{name} done"

# 创建进程
p = Process(target=worker, args=("A",))
p.start()
p.join()

# 进程池
with Pool(4) as pool:
    results = pool.map(worker, ["A", "B", "C", "D"])
    print(results)
```

**进程间通信：**

```python
from multiprocessing import Process, Queue, Pipe, Value, Array
import ctypes

# 1. Queue：进程安全队列
def producer(q):
    for i in range(5):
        q.put(i)

def consumer(q):
    while True:
        item = q.get()
        if item is None:
            break
        print(f"消费: {item}")

q = Queue()
p1 = Process(target=producer, args=(q,))
p2 = Process(target=consumer, args=(q,))
p1.start()
p2.start()
p1.join()
q.put(None)  # 发送结束信号
p2.join()

# 2. Pipe：双向通信管道
def sender(conn):
    conn.send("hello from sender")
    conn.close()

parent_conn, child_conn = Pipe()
p = Process(target=sender, args=(child_conn,))
p.start()
print(parent_conn.recv())  # hello from sender
p.join()

# 3. Value/Array：共享内存
counter = Value('i', 0)  # 共享整数
arr = Array('d', [0.0, 0.0, 0.0])  # 共享数组

def increment(counter):
    for _ in range(100000):
        with counter.get_lock():
            counter.value += 1

# 4. Manager：共享复杂对象
from multiprocessing import Manager

def worker(shared_dict, shared_list):
    shared_dict['key'] = 'value'
    shared_list.append(1)

with Manager() as manager:
    d = manager.dict()
    l = manager.list()
    p = Process(target=worker, args=(d, l))
    p.start()
    p.join()
    print(d)  # {'key': 'value'}
    print(l)  # [1]
```

**多线程 vs 多进程：**

| 特性 | 多线程 | 多进程 |
|------|--------|--------|
| GIL 受限 | 是 | 否 |
| 内存共享 | 共享 | 独立 |
| 创建开销 | 小 | 大 |
| 通信方式 | 共享变量 | Queue、Pipe、共享内存 |
| 适用场景 | IO 密集型 | CPU 密集型 |
| 数据安全 | 需要锁 | 天然隔离 |

**面试加分点：** Python 3.8+ 的 `multiprocessing.shared_memory` 提供了更高效的共享内存方式。

## 三、GIL 对多线程的影响

### 面试问题

GIL 对 CPU 密集型和 IO 密集型任务分别有什么影响？如何选择并发方案？

### 标准答案

**CPU 密集型任务：**

```python
import threading
import time

def cpu_bound(n):
    total = 0
    for i in range(n):
        total += i * i
    return total

# 单线程
start = time.time()
cpu_bound(10**7)
cpu_bound(10**7)
print(f"单线程: {time.time() - start:.2f}s")  # 约 4s

# 多线程（受 GIL 限制，没有加速）
start = time.time()
t1 = threading.Thread(target=cpu_bound, args=(10**7,))
t2 = threading.Thread(target=cpu_bound, args=(10**7,))
t1.start(); t2.start()
t1.join(); t2.join()
print(f"多线程: {time.time() - start:.2f}s")  # 约 4s

# 多进程（绕过 GIL，真正并行）
from multiprocessing import Pool
start = time.time()
with Pool(2) as pool:
    pool.map(cpu_bound, [10**7, 10**7])
print(f"多进程: {time.time() - start:.2f}s")  # 约 2s
```

**IO 密集型任务：**

```python
import threading
import time
import requests

def io_bound(url):
    requests.get(url)

urls = ["https://example.com"] * 10

# 单线程
start = time.time()
for url in urls:
    io_bound(url)
print(f"单线程: {time.time() - start:.2f}s")  # 约 10s

# 多线程（IO 等待时释放 GIL，有效加速）
start = time.time()
threads = [threading.Thread(target=io_bound, args=(url,)) for url in urls]
for t in threads:
    t.start()
for t in threads:
    t.join()
print(f"多线程: {time.time() - start:.2f}s")  # 约 1s

# 协程（最高效）
import asyncio
import aiohttp

async def async_io_bound(url):
    async with aiohttp.ClientSession() as session:
        async with session.get(url) as resp:
            await resp.text()

async def main():
    tasks = [async_io_bound(url) for url in urls]
    await asyncio.gather(*tasks)

start = time.time()
asyncio.run(main())
print(f"协程: {time.time() - start:.2f}s")  # 约 1s
```

**并发方案选择指南：**

| 任务类型 | 推荐方案 | 原因 |
|---------|---------|------|
| IO 密集型 | 协程 asyncio | 最轻量，单线程高并发 |
| IO 密集型 | 多线程 | 简单，适合现有同步代码 |
| CPU 密集型 | 多进程 | 绕过 GIL，利用多核 |
| 混合型 | 多进程 + 协程 | 进程利用多核，协程处理 IO |

## 四、concurrent.futures

### 面试问题

`concurrent.futures` 是什么？它和直接使用 threading/multiprocessing 有什么区别？

### 标准答案

**定义：** Python 3.2+ 引入的高级并发接口，统一线程池和进程池的 API。

```python
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor, as_completed
import time

def worker(n):
    time.sleep(1)
    return n * n

# 1. ThreadPoolExecutor（线程池）
with ThreadPoolExecutor(max_workers=3) as executor:
    # 提交单个任务
    future = executor.submit(worker, 5)
    print(future.result())  # 25

    # 批量提交
    futures = [executor.submit(worker, i) for i in range(5)]

    # 按完成顺序获取结果
    for future in as_completed(futures):
        print(future.result())

    # map 方式（按提交顺序返回）
    results = executor.map(worker, range(5))
    for r in results:
        print(r)

# 2. ProcessPoolExecutor（进程池）
with ProcessPoolExecutor(max_workers=4) as executor:
    results = list(executor.map(worker, range(10)))
    print(results)
```

**异常处理：**

```python
from concurrent.futures import ThreadPoolExecutor, as_completed

def risky_worker(n):
    if n == 3:
        raise ValueError(f"Error at {n}")
    return n * n

with ThreadPoolExecutor() as executor:
    futures = {executor.submit(risky_worker, i): i for i in range(5)}

    for future in as_completed(futures):
        try:
            result = future.result()
            print(f"成功: {result}")
        except Exception as e:
            print(f"失败: {e}")
```

**超时控制：**

```python
from concurrent.futures import ThreadPoolExecutor, TimeoutError

def slow_worker():
    time.sleep(10)
    return "done"

with ThreadPoolExecutor() as executor:
    future = executor.submit(slow_worker())
    try:
        result = future.result(timeout=3)  # 3 秒超时
    except TimeoutError:
        print("任务超时")
        future.cancel()
```

**优势：**
- 统一的 API（线程池和进程池接口一致）
- 自动管理线程/进程的创建和销毁
- 支持 `as_completed` 按完成顺序获取结果
- 内置超时和异常处理
- 上下文管理器自动清理资源

## 五、Python 的异步编程实战

### 面试问题

如何用 asyncio 实现一个并发的爬虫或 API 调用？

### 标准答案

**异步 HTTP 请求：**

```python
import asyncio
import aiohttp
import time

async def fetch(session, url):
    """异步获取单个 URL"""
    async with session.get(url) as response:
        status = response.status
        data = await response.text()
        return {"url": url, "status": status, "length": len(data)}

async def fetch_all(urls):
    """并发获取多个 URL"""
    async with aiohttp.ClientSession() as session:
        tasks = [fetch(session, url) for url in urls]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        return results

urls = [
    "https://example.com",
    "https://httpbin.org/get",
    "https://jsonplaceholder.typicode.com/posts/1",
]

start = time.time()
results = asyncio.run(fetch_all(urls))
print(f"耗时: {time.time() - start:.2f}s")
for r in results:
    if isinstance(r, Exception):
        print(f"错误: {r}")
    else:
        print(f"{r['url']}: {r['status']}, {r['length']} bytes")
```

**带限流的异步任务：**

```python
import asyncio
import aiohttp

async def fetch_with_semaphore(sem, session, url):
    """使用信号量限制并发数"""
    async with sem:
        async with session.get(url) as resp:
            return await resp.text()

async def main():
    urls = [f"https://example.com/page/{i}" for i in range(100)]
    sem = asyncio.Semaphore(10)  # 最多 10 个并发

    async with aiohttp.ClientSession() as session:
        tasks = [fetch_with_semaphore(sem, session, url) for url in urls]
        results = await asyncio.gather(*tasks)
    return results

asyncio.run(main())
```

**异步数据库操作（以 SQLAlchemy 为例）：**

```python
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession
from sqlalchemy.orm import sessionmaker
from sqlalchemy import select, User

async def get_users():
    engine = create_async_engine("postgresql+asyncpg://user:pass@localhost/db")
    async_session = sessionmaker(engine, class_=AsyncSession)

    async with async_session() as session:
        result = await session.execute(select(User))
        users = result.scalars().all()
        return users

asyncio.run(get_users())
```

**异步生成器：**

```python
async def async_range(n):
    for i in range(n):
        await asyncio.sleep(0.1)
        yield i

async def main():
    async for num in async_range(10):
        print(num)

asyncio.run(main())
```

**面试加分点：** 在异步代码中不要使用同步阻塞操作（如 `time.sleep()`、`requests.get()`），这会阻塞整个事件循环。应该使用异步版本（`asyncio.sleep()`、`aiohttp`）或用 `loop.run_in_executor()` 包装。
