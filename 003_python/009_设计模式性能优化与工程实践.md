## 一、Python 的设计模式

### 面试问题

Python 中常用的设计模式有哪些？请举例说明。

### 标准答案

**创建型模式：**

```python
# 1. 单例模式（已在 OOP 文件中详细解答）
# 确保一个类只有一个实例

# 2. 工厂模式：根据参数创建不同类型的对象
class Dog:
    def speak(self):
        return "Woof!"

class Cat:
    def speak(self):
        return "Meow!"

class AnimalFactory:
    @staticmethod
    def create(animal_type):
        if animal_type == "dog":
            return Dog()
        elif animal_type == "cat":
            return Cat()
        raise ValueError(f"Unknown animal: {animal_type}")

animal = AnimalFactory.create("dog")
print(animal.speak())  # Woof!

# 3. 建造者模式：分步创建复杂对象
class QueryBuilder:
    def __init__(self):
        self._table = None
        self._conditions = []
        self._order_by = None
        self._limit = None

    def table(self, table):
        self._table = table
        return self  # 返回 self 支持链式调用

    def where(self, condition):
        self._conditions.append(condition)
        return self

    def order_by(self, field, desc=False):
        self._order_by = f"{field} {'DESC' if desc else 'ASC'}"
        return self

    def limit(self, n):
        self._limit = n
        return self

    def build(self):
        sql = f"SELECT * FROM {self._table}"
        if self._conditions:
            sql += " WHERE " + " AND ".join(self._conditions)
        if self._order_by:
            sql += f" ORDER BY {self._order_by}"
        if self._limit:
            sql += f" LIMIT {self._limit}"
        return sql

query = (QueryBuilder()
         .table("users")
         .where("age > 18")
         .where("status = 'active'")
         .order_by("name")
         .limit(10)
         .build())
print(query)
# SELECT * FROM users WHERE age > 18 AND status = 'active' ORDER BY name ASC LIMIT 10
```

**结构型模式：**

```python
# 1. 代理模式：为对象提供代理以控制访问
class Database:
    def query(self, sql):
        return f"执行: {sql}"

class CachedDatabase:
    def __init__(self, db):
        self._db = db
        self._cache = {}

    def query(self, sql):
        if sql in self._cache:
            print(f"缓存命中: {sql}")
            return self._cache[sql]
        result = self._db.query(sql)
        self._cache[sql] = result
        return result

db = CachedDatabase(Database())
db.query("SELECT * FROM users")  # 执行
db.query("SELECT * FROM users")  # 缓存命中

# 2. 适配器模式：将不兼容的接口转换为兼容的接口
class OldPrinter:
    def print_text(self, text):
        return f"Old: {text}"

class NewPrinter:
    def print_document(self, doc):
        return f"New: {doc}"

class PrinterAdapter:
    def __init__(self, printer):
        self._printer = printer

    def print_text(self, text):
        if hasattr(self._printer, 'print_text'):
            return self._printer.print_text(text)
        return self._printer.print_document(text)

# 3. 装饰器模式：动态地给对象添加职责
class TextProcessor:
    def process(self, text):
        return text

class UpperDecorator:
    def __init__(self, processor):
        self._processor = processor

    def process(self, text):
        return self._processor.process(text).upper()

class TrimDecorator:
    def __init__(self, processor):
        self._processor = processor

    def process(self, text):
        return self._processor.process(text).strip()

processor = TrimDecorator(UpperDecorator(TextProcessor()))
print(processor.process("  hello  "))  # HELLO
```

**行为型模式：**

```python
# 1. 观察者模式：一对多的依赖关系
class EventEmitter:
    def __init__(self):
        self._listeners = {}

    def on(self, event, callback):
        self._listeners.setdefault(event, []).append(callback)

    def emit(self, event, *args, **kwargs):
        for callback in self._listeners.get(event, []):
            callback(*args, **kwargs)

emitter = EventEmitter()
emitter.on("user_login", lambda user: print(f"{user} 登录了"))
emitter.on("user_login", lambda user: print(f"发送欢迎邮件给 {user}"))
emitter.emit("user_login", "Alice")

# 2. 策略模式：定义算法族，分别封装
class SortStrategy:
    def sort(self, data):
        raise NotImplementedError

class BubbleSort(SortStrategy):
    def sort(self, data):
        arr = data.copy()
        n = len(arr)
        for i in range(n):
            for j in range(0, n-i-1):
                if arr[j] > arr[j+1]:
                    arr[j], arr[j+1] = arr[j+1], arr[j]
        return arr

class QuickSort(SortStrategy):
    def sort(self, data):
        if len(data) <= 1:
            return data
        pivot = data[0]
        left = [x for x in data[1:] if x <= pivot]
        right = [x for x in data[1:] if x > pivot]
        return self.sort(left) + [pivot] + self.sort(right)

class Sorter:
    def __init__(self, strategy: SortStrategy):
        self._strategy = strategy

    def sort(self, data):
        return self._strategy.sort(data)

sorter = Sorter(QuickSort())
print(sorter.sort([3, 1, 4, 1, 5, 9, 2, 6]))

# 3. 迭代器模式：顺序访问聚合对象中的元素
# Python 内置支持，通过 __iter__ 和 __next__ 实现

# 4. 模板方法模式：定义算法骨架，子类实现具体步骤
class DataProcessor:
    def process(self):
        data = self.read_data()
        result = self.transform(data)
        self.save_result(result)

    def read_data(self):
        raise NotImplementedError

    def transform(self, data):
        raise NotImplementedError

    def save_result(self, result):
        raise NotImplementedError

class CSVProcessor(DataProcessor):
    def read_data(self):
        return "CSV data"

    def transform(self, data):
        return data.upper()

    def save_result(self, result):
        print(f"Saved: {result}")
```

## 二、Python 的性能优化

### 面试问题

如何优化 Python 程序的性能？有哪些工具和技巧？

### 标准答案

**1. 算法和数据结构优化：**

```python
# 选择合适的数据结构
# dict 查找 O(1) vs list 查找 O(n)
data_list = list(range(10000))
data_dict = {i: True for i in range(10000)}

# 慢
%timeit 9999 in data_list  # 约 100µs

# 快
%timeit 9999 in data_dict  # 约 100ns（快 1000 倍）

# set 去重和查找
data = [1, 2, 2, 3, 3, 3]
unique = list(set(data))  # [1, 2, 3]
```

**2. 代码级优化：**

```python
# 列表推导式比 for 循环快
# 慢
result = []
for i in range(10000):
    result.append(i * 2)

# 快
result = [i * 2 for i in range(10000)]

# 使用 join() 拼接字符串
# 慢
s = ""
for word in ["hello", "world", "python"]:
    s += word + " "

# 快
s = " ".join(["hello", "world", "python"])

# 避免在循环中使用 . 访问属性
# 慢
for i in range(10000):
    math.sqrt(i)

# 快
sqrt = math.sqrt
for i in range(10000):
    sqrt(i)

# 使用局部变量（查找比全局变量快）
def fast_function():
    local_len = len  # 将全局函数赋给局部变量
    data = list(range(10000))
    return local_len(data)

# 使用 map/filter（某些情况下比列表推导式快）
result = list(map(str, range(10000)))

# 使用生成器减少内存
# 慢（占用大量内存）
data = [i * 2 for i in range(10**7)]

# 快（惰性求值）
data = (i * 2 for i in range(10**7))
```

**3. 并发优化：**

```python
# IO 密集型：使用协程
import asyncio
import aiohttp

async def fetch_all(urls):
    async with aiohttp.ClientSession() as session:
        tasks = [session.get(url) for url in urls]
        return await asyncio.gather(*tasks)

# CPU 密集型：使用多进程
from multiprocessing import Pool

def cpu_bound(n):
    return sum(i * i for i in range(n))

with Pool(4) as pool:
    results = pool.map(cpu_bound, [10**6] * 4)
```

**4. 使用 C 扩展和优化库：**

```python
# NumPy 比纯 Python 快很多
import numpy as np

# 慢
data = list(range(10**6))
result = [x * 2 for x in data]

# 快
data = np.arange(10**6)
result = data * 2  # 底层用 C 实现

# Cython：将 Python 编译为 C
# Numba：JIT 编译器
from numba import jit

@jit(nopython=True)
def fast_sum(n):
    total = 0
    for i in range(n):
        total += i
    return total
```

**6. 常见优化技巧汇总：**

| 场景 | 优化方法 | 效果 |
|------|---------|------|
| 字符串拼接 | `"".join(list)` | 快 10 倍 |
| 查找操作 | `set` 或 `dict` 替代 `list` | 快 100-1000 倍 |
| 循环中的属性访问 | 赋给局部变量 | 快 20-30% |
| 大数据处理 | 使用生成器 | 内存减少 90%+ |
| 数值计算 | 使用 NumPy | 快 10-100 倍 |
| CPU 密集型 | 多进程 | 线性加速 |
| IO 密集型 | 协程/多线程 | 并发处理 |

## 三、Python 的单元测试

### 面试问题

如何编写 Python 单元测试？unittest 和 pytest 有什么区别？

### 标准答案

**unittest（标准库）：**

```python
import unittest

class TestCalculator(unittest.TestCase):
    def setUp(self):
        """每个测试方法前执行"""
        self.data = [1, 2, 3, 4, 5]

    def tearDown(self):
        """每个测试方法后执行"""
        pass

    @classmethod
    def setUpClass(cls):
        """整个类前执行一次"""
        print("setUpClass")

    @classmethod
    def tearDownClass(cls):
        """整个类后执行一次"""
        print("tearDownClass")

    def test_sum(self):
        self.assertEqual(sum(self.data), 15)

    def test_max(self):
        self.assertEqual(max(self.data), 5)

    def test_type(self):
        self.assertIsInstance(self.data, list)

    def test_contains(self):
        self.assertIn(3, self.data)

    def test_raises(self):
        with self.assertRaises(ZeroDivisionError):
            1 / 0

    def test_almost_equal(self):
        self.assertAlmostEqual(0.1 + 0.2, 0.3, places=1)

if __name__ == '__main__':
    unittest.main()
```

**pytest（推荐）：**

```python
# test_calculator.py
import pytest

# 基本测试
def test_sum():
    assert sum([1, 2, 3]) == 6

def test_max():
    assert max([1, 2, 3]) == 3

# 异常测试
def test_zero_division():
    with pytest.raises(ZeroDivisionError):
        1 / 0

# fixture：测试夹具
@pytest.fixture
def sample_data():
    return [1, 2, 3, 4, 5]

def test_with_fixture(sample_data):
    assert sum(sample_data) == 15

# 参数化测试
@pytest.mark.parametrize("input,expected", [
    ([1, 2, 3], 6),
    ([0, 0, 0], 0),
    ([-1, 1], 0),
])
def test_sum_parametrize(input, expected):
    assert sum(input) == expected

# 标记测试
@pytest.mark.slow
def test_slow():
    import time
    time.sleep(5)

# 跳过测试
@pytest.mark.skip(reason="not implemented yet")
def test_skip():
    pass

@pytest.mark.skipif(sys.platform == "win32", reason="Linux only")
def test_linux_only():
    pass

# 预期失败
@pytest.mark.xfail(reason="known bug")
def test_known_bug():
    assert 1 == 2
```

**pytest 的 fixture 高级用法：**

```python
import pytest

# scope: function(默认), class, module, session
@pytest.fixture(scope="module")
def db_connection():
    conn = create_connection()
    yield conn  # yield 之前是 setup，之后是 teardown
    conn.close()

# 自动使用
@pytest.fixture(autouse=True)
def reset_state():
    # 每个测试前自动执行
    global_state.clear()
    yield
    # 每个测试后自动执行

# fixture 的依赖
@pytest.fixture
def user(db_connection):
    return db_connection.create_user("test")

def test_user_name(user):
    assert user.name == "test"
```

**pytest vs unittest：**

| 特性 | pytest | unittest |
|------|--------|---------|
| 断言 | `assert` 语句 | `self.assertEqual()` 等 |
| fixture | `@pytest.fixture` | `setUp/tearDown` |
| 参数化 | `@pytest.mark.parametrize` | 需要第三方库 |
| 插件生态 | 丰富 | 有限 |
| 自动发现 | 自动 | 需要 `TestLoader` |
| 学习曲线 | 低 | 中 |

## 五、Python 的 WSGI 和 ASGI

### 面试问题

什么是 WSGI 和 ASGI？它们有什么区别？

### 标准答案

**WSGI（Web Server Gateway Interface）：**

```python
# WSGI 应用就是一个可调用对象
def application(environ, start_response):
    """最简单的 WSGI 应用"""
    status = '200 OK'
    headers = [('Content-Type', 'text/plain')]
    start_response(status, headers)
    return [b'Hello, World!']

# Flask 就是一个 WSGI 框架
from flask import Flask
app = Flask(__name__)

@app.route('/')
def hello():
    return 'Hello, World!'

# 运行：gunicorn app:app
```

**ASGI（Asynchronous Server Gateway Interface）：**

```python
# ASGI 应用支持异步
async def application(scope, receive, send):
    """最简单的 ASGI 应用"""
    assert scope['type'] == 'http'
    await send({
        'type': 'http.response.start',
        'status': 200,
        'headers': [[b'content-type', b'text/plain']],
    })
    await send({
        'type': 'http.response.body',
        'body': b'Hello, World!',
    })

# FastAPI 是一个 ASGI 框架
from fastapi import FastAPI
app = FastAPI()

@app.get('/')
async def hello():
    return {'message': 'Hello, World!'}

# 运行：uvicorn app:app
```

**WSGI vs ASGI：**

| 特性 | WSGI | ASGI |
|------|------|------|
| 同步/异步 | 同步 | 异步 |
| WebSocket | 不支持 | 支持 |
| HTTP/2 | 不支持 | 支持 |
| 长连接 | 不支持 | 支持 |
| 框架 | Flask, Django | FastAPI, Starlette, Django 3.0+ |
| 服务器 | Gunicorn, uWSGI | Uvicorn, Daphne |

**常见部署方案：**

```bash
# WSGI
gunicorn -w 4 -b 0.0.0.0:8000 app:app

# ASGI
uvicorn app:app --host 0.0.0.0 --port 8000 --workers 4

# Nginx + Gunicorn/Uvicorn（生产环境）
# Nginx 作为反向代理，处理静态文件和负载均衡
# Gunicorn/Uvicorn 作为应用服务器
```

## 六、Python 代码规范

Python 有哪些代码规范？如何保证代码质量？

### 标准答案

**PEP 8 核心规范：**

```python
# 1. 缩进：4 个空格
def foo():
    pass

# 2. 行长度：最多 79 字符（代码），72 字符（注释/文档）

# 3. 命名规范
variable_name = 1      # 变量：小写+下划线
function_name = None   # 函数：小写+下划线
ClassName = None       # 类：大驼峰
CONSTANT = 1           # 常量：大写+下划线
_private = 1           # 私有：单下划线前缀
__name = 1             # 名称修饰：双下划线前缀

# 4. 空行
# 顶层函数和类之间：2 个空行
# 类内方法之间：1 个空行

# 5. 导入顺序
# 标准库 → 第三方库 → 本地模块
import os
import sys

import requests
import flask

from myapp import utils
from myapp.models import User
```
