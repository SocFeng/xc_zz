## 一、Python 的魔术方法（双下方法）

### 面试问题

Python 中有哪些常用的魔术方法？请举例说明它们的作用。

### 标准答案

魔术方法是 Python 中以双下划线开头和结尾的特殊方法，用于实现类的内置行为。

**创建和销毁：**

```python
class MyClass:
    def __new__(cls, *args, **kwargs):
        """创建实例，在 __init__ 之前调用"""
        print("__new__ called")
        instance = super().__new__(cls)
        return instance

    def __init__(self, value):
        """初始化实例"""
        print("__init__ called")
        self.value = value

    def __del__(self):
        """实例被垃圾回收时调用"""
        print(f"{self.value} is being deleted")

obj = MyClass(42)
# __new__ called
# __init__ called
```

**字符串表示：**

```python
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __str__(self):
        """print() 和 str() 调用，面向用户"""
        return f"Point({self.x}, {self.y})"

    def __repr__(self):
        """交互式环境和 repr() 调用，面向开发者"""
        return f"Point(x={self.x}, y={self.y})"

p = Point(1, 2)
print(p)        # Point(1, 2)  调用 __str__
repr(p)         # Point(x=1, y=2)  调用 __repr__
```

**比较运算：**

```python
class Student:
    def __init__(self, name, score):
        self.name = name
        self.score = score

    def __eq__(self, other):
        return self.score == other.score

    def __lt__(self, other):
        return self.score < other.score

    def __le__(self, other):
        return self.score <= other.score

    # 使用 @functools.total_ordering 只需实现 __eq__ 和 __lt__
```

**算术运算：**

```python
class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __add__(self, other):
        return Vector(self.x + other.x, self.y + other.y)

    def __sub__(self, other):
        return Vector(self.x - other.x, self.y - other.y)

    def __mul__(self, scalar):
        return Vector(self.x * scalar, self.y * scalar)

    def __abs__(self):
        return (self.x**2 + self.y**2) ** 0.5

    def __repr__(self):
        return f"Vector({self.x}, {self.y})"

v1 = Vector(1, 2)
v2 = Vector(3, 4)
print(v1 + v2)   # Vector(4, 6)
print(v1 * 3)    # Vector(3, 6)
print(abs(v1))   # 2.236...
```

**容器相关：**

```python
class MyList:
    def __init__(self):
        self._data = []

    def __len__(self):
        return len(self._data)

    def __getitem__(self, index):
        return self._data[index]

    def __setitem__(self, index, value):
        self._data[index] = value

    def __delitem__(self, index):
        del self._data[index]

    def __contains__(self, item):
        return item in self._data

    def __iter__(self):
        return iter(self._data)

    def __append__(self, item):
        self._data.append(item)

lst = MyList()
lst._data = [1, 2, 3]
print(len(lst))      # 3  调用 __len__
print(lst[0])        # 1  调用 __getitem__
print(2 in lst)      # True  调用 __contains__
```

**上下文管理：**

```python
class Timer:
    def __enter__(self):
        import time
        self.start = time.time()
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        self.elapsed = time.time() - self.start
        print(f"耗时: {self.elapsed:.4f}s")
        return False  # 不抑制异常

with Timer() as t:
    import time
    time.sleep(1)
# 耗时: 1.0012s
```

**可调用：**

```python
class Multiplier:
    def __init__(self, factor):
        self.factor = factor

    def __call__(self, x):
        return x * self.factor

double = Multiplier(2)
print(double(5))   # 10  实例像函数一样被调用
print(callable(double))  # True
```

## 二、Python 的元类

### 面试问题

什么是元类？`type` 和元类有什么关系？什么时候需要自定义元类？

### 标准答案

**元类的定义：** 元类是创建类的类。`type` 是 Python 内置的元类。

```python
# type 的两种用法
# 1. 查看类型
print(type(42))        # <class 'int'>
print(type("hello"))   # <class 'str'>

# 2. 动态创建类
# type(name, bases, dict)
MyClass = type('MyClass', (object,), {'x': 42})
print(MyClass.x)  # 42
```

**类的创建流程：**

```python
# 当你写：
class Foo(metaclass=type):
    pass

# Python 实际执行的是：
Foo = type('Foo', (), {})

# type.__new__() 创建类对象
# type.__init__() 初始化类对象
```

**自定义元类：**

```python
class SingletonMeta(type):
    """单例元类"""
    _instances = {}

    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            cls._instances[cls] = super().__call__(*args, **kwargs)
        return cls._instances[cls]

class Database(metaclass=SingletonMeta):
    def __init__(self):
        print("初始化数据库连接")

db1 = Database()  # 初始化数据库连接
db2 = Database()  # 不会再次初始化
print(db1 is db2)  # True
```

**验证属性的元类：**

```python
class ValidatedMeta(type):
    def __new__(mcs, name, bases, namespace):
        # 验证所有方法名不能以 _ 开头（举例）
        for key, value in namespace.items():
            if callable(value) and key.startswith('_') and not key.startswith('__'):
                raise ValueError(f"方法 {key} 不能以单下划线开头")
        return super().__new__(mcs, name, bases, namespace)

class MyClass(metaclass=ValidatedMeta):
    def _private(self):  # 会报错
        pass
```

**ORM 中的应用（Django Model）：**

```python
# Django 的 Model 就是通过元类实现的
class ModelMeta(type):
    def __new__(mcs, name, bases, namespace):
        fields = {}
        for key, value in namespace.items():
            if isinstance(value, Field):
                fields[key] = value
        namespace['_fields'] = fields
        return super().__new__(mcs, name, bases, namespace)

class Model(metaclass=ModelMeta):
    pass

class User(Model):
    name = CharField(max_length=100)
    age = IntegerField()
# User._fields 会自动收集所有字段
```

**面试加分点：** 大多数情况下不需要自定义元类，使用类装饰器或 `__init_subclass__` 就够了。元类是"深层魔法"，只在框架开发时使用。

## 三、Python 的描述符

### 面试问题

什么是描述符？`property` 的底层原理是什么？

### 标准答案

**描述符的定义：** 实现了 `__get__`、`__set__`、`__delete__` 中任意一个方法的对象。

```python
class Descriptor:
    """数据描述符（实现了 __get__ 和 __set__）"""
    def __get__(self, instance, owner):
        print(f"__get__ called, instance={instance}, owner={owner}")
        return instance.__dict__.get('_value', 0)

    def __set__(self, instance, value):
        print(f"__set__ called, value={value}")
        if not isinstance(value, (int, float)):
            raise TypeError("值必须是数字")
        instance.__dict__['_value'] = value

    def __delete__(self, instance):
        print(f"__delete__ called")
        del instance.__dict__['_value']

class MyClass:
    value = Descriptor()  # 描述符是类属性

obj = MyClass()
obj.value = 42      # __set__ called, value=42
print(obj.value)    # __get__ called
del obj.value       # __delete__ called
```

**属性查找顺序：**

```python
# 1. 数据描述符（优先级最高）
# 2. 实例属性（__dict__）
# 3. 非数据描述符（只有 __get__）

class DataDesc:
    """数据描述符"""
    def __get__(self, instance, owner):
        return "data descriptor"

    def __set__(self, instance, value):
        pass

class NonDataDesc:
    """非数据描述符"""
    def __get__(self, instance, owner):
        return "non-data descriptor"

class MyClass:
    data = DataDesc()
    nondata = NonDataDesc()

obj = MyClass()
obj.__dict__['data'] = "instance value"
obj.__dict__['nondata'] = "instance value"

print(obj.data)     # data descriptor（数据描述符优先）
print(obj.nondata)  # instance value（实例属性优先）
```

**`property` 的实现原理：**

```python
class property:
    """property 就是一个描述符"""
    def __init__(self, fget=None, fset=None, fdel=None, doc=None):
        self.fget = fget
        self.fset = fset
        self.fdel = fdel
        self.__doc__ = doc

    def __get__(self, instance, owner):
        if instance is None:
            return self
        return self.fget(instance)

    def __set__(self, instance, value):
        self.fset(instance, value)

    def __delete__(self, instance):
        self.fdel(instance)

    def getter(self, fget):
        return type(self)(fget, self.fset, self.fdel, self.__doc__)

    def setter(self, fset):
        return type(self)(self.fget, fset, self.fdel, self.__doc__)

    def deleter(self, fdel):
        return type(self)(self.fget, self.fset, fdel, self.__doc__)
```

**`classmethod` 和 `staticmethod` 的实现也是描述符：**

```python
class classmethod:
    def __init__(self, func):
        self.func = func

    def __get__(self, instance, owner):
        def method(*args, **kwargs):
            return self.func(owner, *args, **kwargs)
        return method
```

## 四、Python 的 MRO（方法解析顺序）

### 面试问题

什么是 MRO？Python 是如何解决多继承中的菱形继承问题的？

### 标准答案

**MRO（Method Resolution Order）：** 多继承时方法的查找顺序。

```python
class A:
    def hello(self):
        return "A"

class B(A):
    def hello(self):
        return "B"

class C(A):
    def hello(self):
        return "C"

class D(B, C):
    pass

d = D()
print(d.hello())  # B
print(D.__mro__)
# (<class 'D'>, <class 'B'>, <class 'C'>, <class 'A'>, <class 'object'>)
```

**C3 线性化算法（Python 3）：**

规则：
1. 子类在父类前面
2. 多个父类保持定义时的顺序
3. 如果一个类出现在继承链中多次，只有最后出现的那次有效

```python
class A:
    pass

class B(A):
    pass

class C(A):
    pass

class D(B, C):
    pass

class E(C, B):
    pass

# D 的 MRO：D -> B -> C -> A -> object
# E 的 MRO：E -> C -> B -> A -> object

# 检查 MRO
print(D.__mro__)
print(D.mro())  # 返回列表
```

**super() 与 MRO：**

```python
class A:
    def method(self):
        return "A"

class B(A):
    def method(self):
        return "B -> " + super().method()

class C(A):
    def method(self):
        return "C -> " + super().method()

class D(B, C):
    def method(self):
        return "D -> " + super().method()

d = D()
print(d.method())
# D -> B -> C -> A
# super() 按照 MRO 顺序调用下一个类的方法
```

**面试加分点：** `super()` 不是调用父类，而是调用 MRO 中的下一个类。在多继承中，这两者是不同的。

## 五、Python 的单例模式

### 面试问题

如何在 Python 中实现单例模式？有哪些实现方式？

### 标准答案

**方式一：使用 `__new__`：**

```python
class Singleton:
    _instance = None

    def __new__(cls, *args, **kwargs):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance

    def __init__(self, value=None):
        self.value = value

a = Singleton(1)
b = Singleton(2)
print(a is b)      # True
print(a.value)     # 2（__init__ 被调用了两次）
```

**方式二：使用装饰器：**

```python
def singleton(cls):
    instances = {}
    def get_instance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    return get_instance

@singleton
class Database:
    def __init__(self):
        print("初始化数据库连接")

db1 = Database()
db2 = Database()
print(db1 is db2)  # True
```

**方式三：使用模块：**

```python
# config.py
class Config:
    def __init__(self):
        self.debug = False
        self.database_url = "sqlite:///db.sqlite"

config = Config()  # 模块级别的实例

# 其他文件中
from config import config  # 模块本身就是单例
```

**方式四：使用元类：**

```python
class SingletonMeta(type):
    _instances = {}

    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            cls._instances[cls] = super().__call__(*args, **kwargs)
        return cls._instances[cls]

class Database(metaclass=SingletonMeta):
    def __init__(self):
        print("初始化数据库连接")

db1 = Database()
db2 = Database()
print(db1 is db2)  # True
```

**线程安全版本：**

```python
import threading

class ThreadSafeSingleton:
    _instance = None
    _lock = threading.Lock()

    def __new__(cls, *args, **kwargs):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:  # 双重检查
                    cls._instance = super().__new__(cls)
        return cls._instance
```

**面试加分点：** 在 Python 中，模块本身就是单例，很多时候不需要显式实现单例模式。

## 六、Python 的多态与鸭子类型

### 面试问题

Python 如何实现多态？什么是鸭子类型？

### 标准答案

**多态：** 不同类的对象对同一消息做出不同响应。

```python
class Dog:
    def speak(self):
        return "Woof!"

class Cat:
    def speak(self):
        return "Meow!"

class Duck:
    def speak(self):
        return "Quack!"

def animal_speak(animal):
    print(animal.speak())

# 不同对象，同一方法，不同行为
animal_speak(Dog())   # Woof!
animal_speak(Cat())   # Meow!
animal_speak(Duck())  # Quack!
```

**鸭子类型：**

```python
# "如果它走起来像鸭子，叫起来像鸭子，那它就是鸭子"
# 不关心对象的类型，只关心对象是否有需要的方法

class File:
    def read(self):
        return "file data"

class Network:
    def read(self):
        return "network data"

class Database:
    def read(self):
        return "db data"

def process(source):
    # 不需要检查 source 的类型
    # 只要它有 read() 方法就行
    return source.read()

# 都可以传入
process(File())
process(Network())
process(Database())
```

**Protocol（Python 3.8+）：**

```python
from typing import Protocol

class Drawable(Protocol):
    def draw(self) -> str: ...

class Circle:
    def draw(self) -> str:
        return "Drawing circle"

class Square:
    def draw(self) -> str:
        return "Drawing square"

def render(shape: Drawable):
    print(shape.draw())

render(Circle())  # OK
render(Square())  # OK
```

**内置的鸭子类型协议：**

```python
# 实现了 __len__ 的对象都可以用 len()
class MyCollection:
    def __init__(self, data):
        self.data = data

    def __len__(self):
        return len(self.data)

c = MyCollection([1, 2, 3])
print(len(c))  # 3

# 实现了 __iter__ 的对象都可以用 for 循环
# 实现了 __getitem__ 的对象都可以用索引访问
```

## 七、Python 的抽象基类（ABC）

### 面试问题

什么是抽象基类？什么时候需要使用它？

### 标准答案

**定义：** 不能被实例化的类，用于定义接口规范。

```python
from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self):
        """计算面积"""
        pass

    @abstractmethod
    def perimeter(self):
        """计算周长"""
        pass

    def description(self):
        """非抽象方法，子类可以直接使用"""
        return f"Area: {self.area()}, Perimeter: {self.perimeter()}"

# shape = Shape()  # TypeError: Can't instantiate abstract class

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius

    def area(self):
        import math
        return math.pi * self.radius ** 2

    def perimeter(self):
        return 2 * math.pi * self.radius

c = Circle(5)
print(c.area())        # 78.54
print(c.description())  # Area: 78.54, Perimeter: 31.42
```

**使用场景：**

1. **定义接口规范**：强制子类实现特定方法
2. **类型检查**：`isinstance(obj, Shape)`
3. **框架设计**：定义插件或扩展点

**ABC vs Protocol：**

| 特性 | ABC | Protocol |
|------|-----|----------|
| 需要继承 | 是 | 否（结构化子类型） |
| 运行时检查 | 是 | 是（Python 3.12+） |
| 适用场景 | 强制约束 | 鸭子类型的类型检查 |

## 九、Python 的类型注解（typing）

### 面试问题

Python 的类型注解是什么？有什么作用？如何使用？

### 标准答案

**基本类型注解：**

```python
# 变量注解
name: str = "Alice"
age: int = 25
scores: list[int] = [90, 85, 92]

# 函数注解
def greet(name: str, times: int = 1) -> str:
    return f"Hello, {name}! " * times
```

**容器类型注解：**

```python
from typing import List, Tuple, Dict, Set, Optional, Union, Callable

# 列表
def process(items: List[int]) -> List[str]:
    return [str(i) for i in items]

# 元组（固定长度）
def point() -> Tuple[float, float]:
    return 1.0, 2.0

# 元组（可变长度）
def coords() -> Tuple[int, ...]:
    return (1, 2, 3)

# 字典
def get_user() -> Dict[str, Union[str, int]]:
    return {"name": "Alice", "age": 25}

# 可选类型（可以是 None）
def find(name: str) -> Optional[str]:
    if name:
        return name
    return None

# 联合类型
def process(value: Union[int, str]) -> str:
    return str(value)

# Python 3.10+ 可以用 | 替代 Union
def process(value: int | str) -> str:
    return str(value)

# Callable
def apply(func: Callable[[int, int], int], a: int, b: int) -> int:
    return func(a, b)
```

**泛型：**

```python
from typing import TypeVar, Generic

T = TypeVar('T')

class Stack(Generic[T]):
    def __init__(self) -> None:
        self._items: list[T] = []

    def push(self, item: T) -> None:
        self._items.append(item)

    def pop(self) -> T:
        return self._items.pop()

int_stack: Stack[int] = Stack()
int_stack.push(1)
# int_stack.push("hello")  # mypy 会报错
```

**Protocol（结构化子类型）：**

```python
from typing import Protocol

class Printable(Protocol):
    def print(self) -> str: ...

class Document:
    def print(self) -> str:
        return "printing document"

def print_item(item: Printable) -> None:
    print(item.print())

print_item(Document())  # OK，Document 满足 Printable 协议
```

**常用类型：**

```python
from typing import (
    Any,            # 任意类型
    NoReturn,       # 永不返回（抛异常）
    Final,          # 不可覆盖
    Literal,        # 字面量类型
    TypeGuard,      # 类型守卫
    TypeAlias,      # 类型别名
)

MAX_SIZE: Final = 100  # 不可修改

def is_string(val: object) -> TypeGuard[str]:
    return isinstance(val, str)

def safe_upper(val: object) -> str:
    if is_string(val):
        return val.upper()  # 类型检查器知道 val 是 str
    return str(val).upper()
```

**面试加分点：** 类型注解不会影响运行时性能，只是给类型检查器（mypy、pyright）看的。但可以提高代码可读性和 IDE 的自动补全能力。
